>>> list(itertools.cycle(word(count=3), 3))
['labore', 'tempor', 'commodo']
>>> list(itertools.cycle(word(count=3, func='capitalize'), 3))
['Ea', 'Lorem', 'Do']
>>> list(itertools.cycle(word(count=3, func=lambda s: s.upper()), 3))
['UT', 'AMET', 'EXCEPTEUR']
